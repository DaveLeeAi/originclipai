# DB_SCHEMA_PLAN.md

> Complete PostgreSQL schema for OriginClipAI v1. Managed via Supabase with Row Level Security.

---

## Schema Overview

```
users (Supabase Auth managed)
  └── jobs (1:many)
       ├── transcripts (1:1)
       ├── clips (1:many)
       │    └── scheduled_posts (1:many)
       └── text_outputs (1:many)
            └── scheduled_posts (1:many)
  └── social_connections (1:many)
  └── prompt_templates (1:many)
  └── api_keys (1:many)
```

---

## Tables

### profiles

Extends Supabase auth.users with app-specific data.

```sql
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email TEXT NOT NULL,
  display_name TEXT,
  avatar_url TEXT,
  plan TEXT NOT NULL DEFAULT 'free' CHECK (plan IN ('free', 'creator', 'pro', 'business')),
  stripe_customer_id TEXT,
  stripe_subscription_id TEXT,
  minutes_used_this_cycle INTEGER NOT NULL DEFAULT 0,
  minutes_limit INTEGER NOT NULL DEFAULT 30,
  billing_cycle_start TIMESTAMPTZ,
  default_caption_style TEXT DEFAULT 'karaoke',
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_profiles_stripe ON profiles(stripe_customer_id);
```

### jobs

Central entity. One per ingest action.

```sql
CREATE TYPE source_type AS ENUM (
  'youtube_url', 'video_url', 'video_upload', 
  'audio_upload', 'article_url', 'pdf_upload'
);

CREATE TYPE job_status AS ENUM (
  'created', 'ingesting', 'transcribing', 'analyzing', 
  'rendering', 'complete', 'failed', 'cancelled'
);

CREATE TABLE jobs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  
  -- Source
  source_type source_type NOT NULL,
  source_url TEXT,                          -- For URL-based inputs
  source_file_key TEXT,                     -- S3/storage key for uploads
  source_title TEXT,                        -- Extracted or user-provided
  source_duration_seconds FLOAT,            -- For video/audio sources
  source_metadata JSONB DEFAULT '{}',       -- Flexible: thumbnail_url, author, description, etc.
  
  -- Processing
  status job_status NOT NULL DEFAULT 'created',
  current_step TEXT,                        -- Human-readable current step
  progress JSONB DEFAULT '{}',             -- Per-step progress: {ingest: 'complete', transcribe: 'running', ...}
  error TEXT,                               -- Error message if failed
  
  -- Stats
  clip_count INTEGER DEFAULT 0,
  text_output_count INTEGER DEFAULT 0,
  minutes_consumed FLOAT DEFAULT 0,         -- Billing: minutes charged for this job
  
  -- Timestamps
  processing_started_at TIMESTAMPTZ,
  processing_completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_jobs_user ON jobs(user_id);
CREATE INDEX idx_jobs_status ON jobs(status);
CREATE INDEX idx_jobs_user_created ON jobs(user_id, created_at DESC);
```

### transcripts

One per job (for video/audio sources).

```sql
CREATE TYPE transcription_engine AS ENUM ('whisper', 'assemblyai');

CREATE TABLE transcripts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID NOT NULL UNIQUE REFERENCES jobs(id) ON DELETE CASCADE,
  
  -- Content
  full_text TEXT NOT NULL,                  -- Plain text transcript
  language TEXT DEFAULT 'en',               -- Detected language code
  
  -- Structured data
  segments JSONB NOT NULL DEFAULT '[]',     -- [{start, end, text, speaker_id, confidence}]
  speakers JSONB NOT NULL DEFAULT '[]',     -- [{id, label, role, talk_time_pct, talk_time_seconds}]
  word_timestamps JSONB DEFAULT '[]',       -- [{word, start, end, speaker_id}]
  
  -- Metadata
  engine transcription_engine NOT NULL,
  duration_seconds FLOAT,
  word_count INTEGER,
  confidence_avg FLOAT,                     -- Average transcription confidence
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_transcripts_job ON transcripts(job_id);
```

### clips

Video clip candidates generated by the analyze step.

```sql
CREATE TYPE clip_status AS ENUM ('review', 'approved', 'rejected');
CREATE TYPE speaker_role AS ENUM ('host', 'guest', 'solo', 'unknown');

CREATE TABLE clips (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  
  -- Timing
  start_time FLOAT NOT NULL,               -- Seconds from source start
  end_time FLOAT NOT NULL,                 -- Seconds from source start
  duration FLOAT NOT NULL,                 -- end_time - start_time
  
  -- AI-generated content
  title TEXT NOT NULL,                      -- AI-generated clip title
  hook TEXT,                                -- Opening hook text (first sentence)
  transcript_excerpt TEXT NOT NULL,         -- Transcript text for this clip
  
  -- Scoring
  score INTEGER NOT NULL CHECK (score >= 0 AND score <= 100),
  score_factors JSONB DEFAULT '{}',         -- {coherence, hook_strength, topic_clarity, emotional_energy}
  
  -- Speaker
  primary_speaker_id TEXT,                  -- Speaker who talks most in this clip
  speaker_role speaker_role DEFAULT 'unknown',
  speakers_present JSONB DEFAULT '[]',      -- All speakers in this clip [{id, talk_pct}]
  
  -- Render state
  status clip_status NOT NULL DEFAULT 'review',
  caption_style TEXT DEFAULT 'karaoke',
  rendered_files JSONB DEFAULT '{}',        -- {"9x16": "storage-key", "1x1": "storage-key", "16x9": "storage-key"}
  render_status TEXT DEFAULT 'pending',     -- pending, rendering, complete, failed
  
  -- Social
  social_caption TEXT,                      -- AI-generated social caption for this clip
  hashtags JSONB DEFAULT '[]',              -- ["#content", "#creator", ...]
  
  -- Ordering
  sort_order INTEGER NOT NULL DEFAULT 0,    -- Display order (by score desc)
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_clips_job ON clips(job_id);
CREATE INDEX idx_clips_job_status ON clips(job_id, status);
CREATE INDEX idx_clips_job_score ON clips(job_id, score DESC);
```

### text_outputs

Generated text content pieces.

```sql
CREATE TYPE text_output_type AS ENUM (
  'linkedin_post', 'x_thread', 'newsletter_section',
  'summary', 'chapter_markers', 'social_caption', 'blog_draft',
  'show_notes', 'custom'
);

CREATE TYPE text_output_status AS ENUM ('draft', 'approved', 'scheduled', 'posted');

CREATE TABLE text_outputs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
  
  -- Content
  type text_output_type NOT NULL,
  label TEXT NOT NULL,                      -- Human-readable: "LinkedIn Post", "X Thread (7 posts)"
  content TEXT NOT NULL,                    -- The generated text
  word_count INTEGER NOT NULL DEFAULT 0,
  
  -- For X threads specifically
  thread_posts JSONB,                       -- [{post_number, text}] — split thread into individual posts
  
  -- State
  status text_output_status NOT NULL DEFAULT 'draft',
  
  -- Metadata
  prompt_template_id UUID,                  -- If generated from custom template
  metadata JSONB DEFAULT '{}',             -- Flexible metadata per type
  
  -- Ordering
  sort_order INTEGER NOT NULL DEFAULT 0,
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_text_outputs_job ON text_outputs(job_id);
CREATE INDEX idx_text_outputs_job_type ON text_outputs(job_id, type);
```

### scheduled_posts

Posts queued for social platform publishing.

```sql
CREATE TYPE platform AS ENUM ('youtube', 'tiktok', 'linkedin', 'x', 'instagram', 'facebook');
CREATE TYPE post_status AS ENUM ('queued', 'posting', 'posted', 'failed', 'cancelled');

CREATE TABLE scheduled_posts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  
  -- Content reference (one of these is set)
  clip_id UUID REFERENCES clips(id) ON DELETE SET NULL,
  text_output_id UUID REFERENCES text_outputs(id) ON DELETE SET NULL,
  
  -- Target
  platform platform NOT NULL,
  social_connection_id UUID REFERENCES social_connections(id),
  
  -- Schedule
  scheduled_at TIMESTAMPTZ NOT NULL,
  
  -- State
  status post_status NOT NULL DEFAULT 'queued',
  platform_post_id TEXT,                    -- ID returned by platform after posting
  platform_post_url TEXT,                   -- URL of published post
  error TEXT,                               -- Error message if failed
  retry_count INTEGER DEFAULT 0,
  last_retry_at TIMESTAMPTZ,
  
  -- Content snapshot (in case source is deleted)
  content_snapshot JSONB,                   -- Snapshot of content at schedule time
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  
  CHECK (clip_id IS NOT NULL OR text_output_id IS NOT NULL)
);

CREATE INDEX idx_scheduled_posts_user ON scheduled_posts(user_id);
CREATE INDEX idx_scheduled_posts_status_time ON scheduled_posts(status, scheduled_at)
  WHERE status = 'queued';
CREATE INDEX idx_scheduled_posts_clip ON scheduled_posts(clip_id);
CREATE INDEX idx_scheduled_posts_text ON scheduled_posts(text_output_id);
```

### social_connections

OAuth connections to social platforms.

```sql
CREATE TABLE social_connections (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  
  platform platform NOT NULL,
  platform_user_id TEXT,                    -- User's ID on the platform
  platform_username TEXT,                   -- Display name
  platform_avatar_url TEXT,
  
  -- OAuth tokens (encrypted at rest)
  access_token TEXT NOT NULL,
  refresh_token TEXT,
  token_expires_at TIMESTAMPTZ,
  scopes JSONB DEFAULT '[]',
  
  -- Status
  is_active BOOLEAN NOT NULL DEFAULT true,
  last_used_at TIMESTAMPTZ,
  error TEXT,                               -- Last error if connection is broken
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  
  UNIQUE(user_id, platform)                 -- One connection per platform per user
);

CREATE INDEX idx_social_connections_user ON social_connections(user_id);
```

### prompt_templates

User-defined prompt templates for custom text outputs.

```sql
CREATE TABLE prompt_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  
  name TEXT NOT NULL,                       -- "My LinkedIn Style"

  usage_count INTEGER DEFAULT 0,
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_prompt_templates_user ON prompt_templates(user_id);
```

### api_keys

API authentication keys for headless access.

```sql
CREATE TABLE api_keys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  
  key_prefix TEXT NOT NULL,                 -- First 8 chars for display: "sk_live_a1b2..."
  key_hash TEXT NOT NULL,                   -- bcrypt hash of full key
  name TEXT NOT NULL DEFAULT 'Default',     -- User-assigned name
  
  is_active BOOLEAN NOT NULL DEFAULT true,
  last_used_at TIMESTAMPTZ,
  usage_count INTEGER DEFAULT 0,
  
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  revoked_at TIMESTAMPTZ
);

CREATE INDEX idx_api_keys_user ON api_keys(user_id);
CREATE INDEX idx_api_keys_prefix ON api_keys(key_prefix);
```

## Row Level Security Policies

All tables use RLS. Users can only access their own data.

```sql
-- Enable RLS on all tables
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE jobs ENABLE ROW LEVEL SECURITY;
ALTER TABLE transcripts ENABLE ROW LEVEL SECURITY;
ALTER TABLE clips ENABLE ROW LEVEL SECURITY;
ALTER TABLE text_outputs ENABLE ROW LEVEL SECURITY;
ALTER TABLE scheduled_posts ENABLE ROW LEVEL SECURITY;
ALTER TABLE social_connections ENABLE ROW LEVEL SECURITY;
ALTER TABLE prompt_templates ENABLE ROW LEVEL SECURITY;
ALTER TABLE api_keys ENABLE ROW LEVEL SECURITY;

-- Profile: users see own profile
CREATE POLICY "Users see own profile" ON profiles FOR ALL USING (auth.uid() = id);

-- Jobs: users see own jobs
CREATE POLICY "Users see own jobs" ON jobs FOR ALL USING (auth.uid() = user_id);

-- Transcripts: via job ownership
CREATE POLICY "Users see own transcripts" ON transcripts FOR ALL
  USING (EXISTS (SELECT 1 FROM jobs WHERE jobs.id = transcripts.job_id AND jobs.user_id = auth.uid()));

-- Clips: via job ownership
CREATE POLICY "Users see own clips" ON clips FOR ALL
  USING (EXISTS (SELECT 1 FROM jobs WHERE jobs.id = clips.job_id AND jobs.user_id = auth.uid()));

-- Text outputs: via job ownership
CREATE POLICY "Users see own text_outputs" ON text_outputs FOR ALL
  USING (EXISTS (SELECT 1 FROM jobs WHERE jobs.id = text_outputs.job_id AND jobs.user_id = auth.uid()));

-- Scheduled posts: user ownership
CREATE POLICY "Users see own scheduled_posts" ON scheduled_posts FOR ALL USING (auth.uid() = user_id);

-- Social connections: user ownership
CREATE POLICY "Users see own connections" ON social_connections FOR ALL USING (auth.uid() = user_id);

-- Prompt templates: user ownership
CREATE POLICY "Users see own templates" ON prompt_templates FOR ALL USING (auth.uid() = user_id);

-- API keys: user ownership
CREATE POLICY "Users see own api_keys" ON api_keys FOR ALL USING (auth.uid() = user_id);
```

## Triggers

```sql
-- Auto-update updated_at on modification
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_profiles_updated BEFORE UPDATE ON profiles FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_jobs_updated BEFORE UPDATE ON jobs FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_clips_updated BEFORE UPDATE ON clips FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_text_outputs_updated BEFORE UPDATE ON text_outputs FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_scheduled_posts_updated BEFORE UPDATE ON scheduled_posts FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_social_connections_updated BEFORE UPDATE ON social_connections FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```
